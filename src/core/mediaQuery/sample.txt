 if (preceedRule) {
        observable = createObservable<any>(function (next) {
          ruleTrack.observable.observe(function (value) {
            next([
              { current: preceedRule.state, last: preceedRule.state },
              { current: value, last: rule.state },
            ]);
          });
          preceedRule.observable.observe(function (value) {
            next([
              { current: rule.state, last: rule.state },
              { current: value, last: preceedRule.state },
            ]);
          });
        });

        observable.observe(function (value: [RuleHistory, RuleTrack]) {
          rule.state = value;
          if (rule.scope) {
            rule.scope(value);
          }
        });
